        <h2>忘记密码</h2>
        <form id="forgotPasswordForm" action="#" method="post" onsubmit="return sendResetLink()">
            <input type="email" id="forgotPasswordEmail" name="email" placeholder="注册邮箱" required>
            <button type="submit">发送重置链接</button>

<script>
    // 用户数据
    var users = [
        { username: "user1", email: "user1@example.com", password: "password1" },
        { username: "user2", email: "user2@example.com", password: "password2" }
    ];

    // 验证登录
    function validateLogin() {
        var username = document.getElementById("loginUsername").value;
        var password = document.getElementById("loginPassword").value;

        // 查找用户
        var foundUser = users.find(function(user) {
            return user.username === username && user.password === password;
        });

        if (foundUser) {
            alert("登录成功！");
            // 在此处记录用户登录信息
            console.log("用户 " + foundUser.username + " 登录成功！");
            return true;
        } else {
            alert("用户名或密码错误，请重试！");
            return false;
        }
    }

    // 用户注册
    function registerUser() {
        var username = document.getElementById("registerUsername").value;
        var email = document.getElementById("registerEmail").value;
        var password = document.getElementById("registerPassword").value;

        // 检查用户名是否已存在
        var existingUser = users.find(function(user) {
            return user.username === username;
        });

        if (existingUser) {
            alert("用户名已存在，请选择其他用户名！");
            return false;
        }

        // 添加新用户
        var newUser = { username: username, email: email, password: password };
        users.push(newUser);
        alert("注册成功！");
        // 在此处记录用户注册信息
        console.log("新用户注册成功：" + JSON.stringify(newUser));
        showPage('loginPage');
        return false;
    }

    // 发送密码重置链接
    function sendResetLink() {
        var email = document.getElementById("forgotPasswordEmail").value;

        // 检查邮箱是否注册
        var existingUser = users.find(function(user) {
            return user.email === email;
        });

        if (existingUser) {
            alert("重置链接已发送至您的邮箱，请查收！");
            // 在此处发送密码重置链接的逻辑
            console.log("密码重置链接已发送至邮箱：" + email);
            showPage('loginPage');
        } else {
            alert("该邮箱未注册，请确认邮箱地址！");
        }
        return false;
    }

    // 切换页面
    function showPage(pageId) {
        // 隐藏所有页面
        var pages = document.querySelectorAll('.page');
        pages.forEach(function(page) {
            page.style.display = 'none';
        });
        // 显示目标页面
        var targetPage = document.getElementById(pageId);
        if (targetPage) {
            targetPage.style.display = 'block';
        }
    }
</script>
